        \draw (-1,0.1) -- (-1,-0.1) node[xshift=-0.25cm, below, gray] {$-1$};

        % Y ticks
        \draw (0.1,-1) -- (-0.1,-1) node[yshift=-0.2cm, left, gray] {$-1$};
        \draw (0.1,1) -- (-0.1,1) node[yshift=0.2cm, left, gray] {$1$};

        % 2-norm
        \draw[thick, blue] (0,0) circle (1);
        \node[above,black] at (0., 1.5) {$\Vert \mathbf{x} \Vert_2 = 1$};
