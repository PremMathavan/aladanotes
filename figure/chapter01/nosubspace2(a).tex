\documentclass[crop,tikz]{standalone}% 'crop' is the default for v1.0, before it was 'preview'
%\usetikzlibrary{...}% tikz package already loaded by 'tikz' option
\usepackage{amsmath,amsthm,amssymb,mathrsfs,amsfonts,dsfont}

\begin{document}
    \begin{tikzpicture}[scale=0.32]
        % Axes
        \draw[->, gray] (-5,0) -- (5,0) node[right] {$x_1$};
        \draw[->, gray] (0,-5) -- (0,5) node[above] {$x_2$};
        % Axis ticks
        \foreach \x in {-4,-2,2,4}
            \draw (\x,0.1) -- (\x,-0.1) node[below, gray] {$\x$};
        \foreach \y in {-4,-2,2,4}
            \draw (0.1,\y) -- (-0.1,\y) node[left, gray] {$\y$};
        % Plot the parabola
        \draw[thick, blue, domain=-3.1:3.1] plot (\x,{\x*\x / 2});
        % Label
        \node[right, black] at (1.75, 2) {$S = \left\{ \begin{bmatrix} x \\ x^2 \end{bmatrix}\right\}$};
        % Huge font for the label with brown color
        \node[font=\Large, text=brown] at (-4, 4.5) {$\mathbb{R}^2$};
    \end{tikzpicture}
