    % Define the vectors
    \tdplotsetmaincoords{60}{120}
    \begin{tikzpicture}[tdplot_main_coords, scale=2]

        \draw[->, thick, gray] (0,0,-0.8) -- (0,0,0.8);

        % Define the coordinates for the vectors
        \coordinate (O) at (0,0,0);
        \coordinate (V1) at (0.25,0.25,0);
        \coordinate (V2) at (-0.25,0,0.25);
        \coordinate (V_n1) at (-0.5,-0.5,0);
        \coordinate (V_n1_p2) at (-1,-0.5,0.5);

        \filldraw[opacity=0.2, blue!30] (V_n1_p2) -- (V_p1_p2) -- (V_p1_n2) -- (V_n1_n2);

        % Draw the vectors
